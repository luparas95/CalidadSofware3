3   import com.politics.controllers.LoginController;
4   import com.politics.views.validations.Validations;
5   
6   import java.awt.event.*;
7   
8   import javax.swing.*;
9   
10  public class LoginView {
11      
12      public static void mostrar() {
13          
14          final JFrame f = new JFrame("Login");
15  
16          JLabel nombreLbl = new JLabel("Nombre");
17          nombreLbl.setBounds(50, 50, 280, 14);
18          f.getContentPane().add(nombreLbl);
19  
20          final JTextField nombreField = new JTextField();
21          nombreField.setBounds(50, 70, 280, 25);
22          f.getContentPane().add(nombreField);
23          nombreField.setColumns(10);
24           
25          JLabel passwordLbl = new JLabel("Password");
26          passwordLbl.setBounds(50, 150, 280, 14);
27          f.getContentPane().add(passwordLbl);
28  
29          final JPasswordField passwordField = new JPasswordField();
30          passwordField.setBounds(50, 170, 280, 25);
31          f.getContentPane().add(passwordField);
32          passwordField.setColumns(10);
33          
34          JButton loginButton = new JButton("Aceptar");
35          
36          loginButton.setBounds(130, 310, 130, 30);
37          
38          loginButton.addActionListener(new ActionListener() {
39  
40              public void actionPerformed(ActionEvent e) {
41                  
42                  if (!Validations.cadenaValida(nombreField.getText())) {
43                  
44                      JOptionPane.showMessageDialog(f, "Introduzca un nombre");
45                  
46                  } else if (!Validations.cadenaValida(passwordField.getText())) {
47                  
48                      JOptionPane.showMessageDialog(f, "Introduzca un password");
49                  
50                  } else {
51                  
52                      f.setVisible(false);
53                      LoadingView.mostrar();
54                      if (LoginController.checkLogin(nombreField.getText(), String.valueOf(passwordField.getPassword()))) {
55                      
56                          LoadingView.ocultar();
57                          MenuPrincipalView.mostrar();
58                      
59                      } else {
60                      
61                          LoadingView.ocultar();
62                          f.setVisible(true);
63                          JOptionPane.showMessageDialog(f, "Usuario o password incorrecto");
64                      
65                      }
66                  
67                  }
68  
69              }
70  
71          });
72          
73          f.add(loginButton);
74          
75          f.setSize(400, 400);
76          f.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
77          f.setLayout(null);
78          f.setVisible(true);
79    
80      }
81  
82  }
